fun startsWith(o: Object): Boolean {
        if (o !is SimpleArrayList) {
            return false
        }
        if ((o as SimpleArrayList).values!!.size <= values!!.size) {
            for (i in (o as SimpleArrayList).values.indices) {
                if (!(o as SimpleArrayList).values!![i].equals(values[i])) {
                    return false
                }
            }
        } else if ((o as SimpleArrayList).values!!.size > values.size) {
            for (i in values.indices) {
                if (!(o as SimpleArrayList).values!![i].equals(values[i])) {
                    return false
                }
            }
        }
        return true
   }

class LastEquals : SimpleLinkedList() {
    fun lastEquals(o: Object): Boolean {
        var fin: Object? = null
        if (start == null) {
            return false
        }
        var current: Item = start
        while (current != null) {
            if (current.next == null) {
                fin = current.value
            }
            current = current.next
        }
        return o.equals(fin)
    }
}

object ListIsLonger {  
    fun isLonger(x: List<String>, y: List<String>): List<String> {  
        return if (x.size() > y.size()) {  
            x  
        } else {  
            y  
        }  
    }  
}
