Recursion and Graphs

Unweighted Graph Count Positive

import cs1.graphs.UnweightedGraph;
import cs1.graphs.GraphNode;
import java.util.Set;
import java.util.HashSet;

public class GraphCount {
	public static int count(UnweightedGraph<Integer> x) {
		if (x == null) {
			throw new IllegalArgumentException();
		}
		int count = 0;
		Set<GraphNode<Integer>> set = new HashSet<>();
		traverse(x.getNode(), set);
		for (GraphNode<Integer> i: set) {
			if (i.getValue() > 0) {
				count++;
			}
		}
		return count;
	}
	private static void traverse(GraphNode<Integer> x, Set<GraphNode<Integer>> y) {
		if (y.contains(x)) {
			return;
		}
		y.add(x);
		for (GraphNode<Integer> i: x.getNeighbors()) {
			traverse(i, y);
		}
	}
}

Binary Tree Count Children Sum to Odd

import cs125.trees.BinaryTree;
public class BinaryTreeCounter {
	public static int countChildSumToOdd(BinaryTree<Integer> x) {
		 if(x == null) {
			 return 0;
		 } else if (x.getLeft() != null && x.getRight() == null && x.getLeft().getValue() % 2 != 0) {
			 return 1 + countChildSumToOdd(x.getLeft());
		 } else if (x.getRight() != null && x.getLeft() == null && x.getRight().getValue() % 2 != 0) {
			 return 1 + countChildSumToOdd(x.getRight());
		 } else if (x.getRight() != null && x.getLeft() != null && (x.getRight().getValue() + x.getLeft().getValue()) % 2 != 0) {
			 return 1 + countChildSumToOdd(x.getRight()) + countChildSumToOdd(x.getLeft());
	     } else {
		     return countChildSumToOdd(x.getRight()) + countChildSumToOdd(x.getLeft());
		 }
}
