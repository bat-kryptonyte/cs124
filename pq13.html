Merge

public class Merger {
	public static int[] merge(int[] x, int[] y) {
		if (x == null || y == null) {
			throw new IllegalArgumentException();
		}
		if (x.length == 0) {
			return y;
		} else if (y.length == 0) {
			return x;
		}
		int[] toReturn = new int[x.length + y.length];
		int first = 0;
		int second = 0;
		for (int i = 0; i < toReturn.length; i++) {
			if (first == x.length) {
				toReturn[i] = y[second++];
			} else if (second == y.length) {
				toReturn[i] = x[first++];
			} else if (x[first] > y[second]) {
				toReturn[i] = x[first++];
			} else {
				toReturn[i] = y[second++];
			}
		}
		return toReturn;
	}
}

QuickSort Partition by Last Value

public class Partitioner {
	public static int partition(int[] values) {
		if (values == null || values.length == 0) {
			throw new IllegalArgumentException();
		}
		int pivot = values[values.length - 1];
		int p1 = 0;
		for (int i = 0; i < values.length - 1; i++) {
			if (values[i] < pivot) {
				swap(values, i, p1);
				p1++;
			}
		}
		swap(values, p1, values.length - 1);
		return p1;
		}
	}
	private static void swap(int[] a, int b, int c) {
		int temp = a[b];
		a[b] = a[c];
		a[c] = temp;
	}
}
