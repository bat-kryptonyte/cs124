SimpleLinkedList last

public class LastLinkedList extends SimpleLinkedList {
	public Object last() {
		Object toReturn = null;
		for (Item current = start; current != null; current = current.next) {
			toReturn = current.value;
		}
		return toReturn;
	}
}

Binary Tree Count Has Both Children

public class BinaryTreeCounter {
	public static int countHasBothChildren(BinaryTree<?> x) {
		if (x == null) {
			return 0;
		} else if (x.getLeft() != null && x.getRight() != null) {
			return 1 + countHasBothChildren(x.getLeft()) + countHasBothChildren(x.getRight());
		} 
		return countHasBothChildren(x.getLeft()) + countHasBothChildren(x.getRight());
	}
}
