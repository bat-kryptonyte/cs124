Problem 1

public class Comparer {
	private int x;
	public void setValue(int data) {
		x = data;
	}
	public boolean compare(int data) {
		return data > x;
	}
}

Problem 2

public class OddCounter {
	private int count;
	public OddCounter(int data) {
		assert data >= 0;
		count = data;
	}
	public void addValue(int data) {
		if (data % 2 != 0) {
			count++;
		}
	}
	public int getCount() {
		return count;
	}
}

Problem 3

public class Cat {
	private int lives;
	private String name;
	public Cat(String x, int y) {
		assert x != null;
		assert y >= 0;
		lives = y;
		name = x;
	}
	public int getLives() {
		return lives;
	}
	public void setLives(int data) {
		assert data >= 0;
		lives = data;
	}
	public String getName() {
		return name;
	}
}
