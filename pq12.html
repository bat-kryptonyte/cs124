Count Binary Tree Non Leaves

import cs125.trees.BinaryTree;
public class BinaryTreeCounter {
	public static int countNonLeaves(BinaryTree<?> x) {
		if (x == null) {
			return 0;
		} else if (x.getLeft() != null && x.getRight == null) {
			return 1 + countNonLeaves(x.getLeft());
		} else if (x.getRight() != null && x.getLeft == null) {
			return 1 + countNonLeaves(x.getRight());
		} else if (x.getRight() != null && x.getLeft() != null) {
			return 1 + countNonLeaves(x.getRight()) + countNonLeaves(x.getLeft());
		} else {
			return countNonLeaves(x.getRight()) + countNonLeaves(x.getLeft());
		}
	}
}

import cs1.graphs.UnweightedGraph;
import cs1.graphs.GraphNode;
import java.util.Set;
import java.util.HashSet;

public class GraphSum {
	public static int sum(UnweightedGraph<Integer> x) {
		if (x == null) {
			throw new IllegalArgumentException();
		}
		int sum = 0;
		Set<GraphNode<Integer>> set = new HashSet<>();
		travis(x.getNode(), set);
		for (GraphNode<Integer> i: set) {
			if (i.getValue() < 0) {
				sum += i.getValue();
			}
		}
		return sum;
	}
	private static void travis(GraphNode<Integer> x, Set<GraphNode<Integer>> y) {
		if (y.contains(x)) {
			return;
		}
		y.add(x);
		for (GraphNode<Integer> i: x.getNeighbors()) {
			travis(i, y);
		}
	}
}
