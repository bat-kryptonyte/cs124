public class SimpleArrayList {
	private final Object[] values;
	public SimpleArrayList(Object[] setValues) {
		assert setValues != null;
		values = setValues;
	}
	public boolean startsWith(Object o) {
		
			if (!(o instanceof SimpleArrayList && ((SimpleArrayList) o).values.length <= values.length)) {
				return false;
			} else {
				for (int i = 0; i < ((SimpleArrayList) o).values.length; i++) {
					if (!((SimpleArrayList) o).values[i].equals(values[i])) {
						return false;
					}
				}
			}
			return true;
		
	}
}

public class LastEquals extends SimpleLinkedList {

		Object fin = null;
		if (start == null) {
			return false;

				fin = current.value;
			}
		}

List Is Longer

import java.util.List;
public class ListIsLonger {
	public static List<String> isLonger(List<String> x, List<String> y) {
		if (x.size() > y.size()) {
			return x;
		} else {
			return y;
		}
	}
}
