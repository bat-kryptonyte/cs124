Interfaces

Spider

class Spider(private val legs: Int) : OurComparable {  
    init {  
        kotlin.require(legs > 0)  
    }  
  
    operator fun compareTo(a: Any): Int {  
        kotlin.require(a is Spider)  
        if (legs < (a as Spider).legs) {  
            return -1  
		} else if (legs > a.legs) {  
            return 1  
		}  
        return 0  
	}  
}

CountLessThan

object CountLessThan {  
    fun count(x: Array<OurComparable>, y: OurComparable?): Int {  
        assert(x != null)  
        assert(y != null)  
        var count = 0  
		for (i in x) {  
            if (i.compareTo(y) < 0) {  
                count++  
            }  
        }  
        return count  
    }  
}

DifferentCount

object DifferentCount {  
    fun count(a: Array<Any>, b: Any): Int {  
        assert(a != null)  
        var count = 0  
		for (i in a) {  
	        if (!(i === b)) {  
                count++  
            }  
        }  
        return count  
    }  
}
